Query 3: Estudiantes activos que cursaron Matemáticas VII y pasaron Física I (nota >= 4) Temp_Mate7 = PROYECTAR[carnet] ( SELECCIONAR[asignatura = "Matematicas VII"] (Calificacion) ) Temp_Fisica1 = PROYECTAR[carnet] ( SELECCIONAR[asignatura = "Fisica I" AND nota >= 4] (Calificacion) )

Resultado = SELECCIONAR[activo = true] (Estudiante) NATURAL JOIN (Temp_Mate7 INTERSECCION Temp_Fisica1)

Materias_Validas = SELECCIONAR[year <= 2046] (Calificacion) NATURAL JOIN Asignatura
